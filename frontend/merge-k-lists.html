Merge Sort

Sort an array by dividing it into halves, sorting each half, and merging the results. Stable, divide-and-conquer, and runs in O(n log n) time.

def merge_sort(arr):
    if len(arr) <= 1:
        return arr

    mid = len(arr) // 2
    left = merge_sort(arr[:mid])
    right = merge_sort(arr[mid:])

    return merge(left, right)


def merge(left, right):
    result = []
    i = j = 0

    while i < len(left) and j < len(right):
        if left[i] <= right[j]:
            result.append(left[i])
            i += 1
        else:
            result.append(right[j])
            j += 1

    result.extend(left[i:])
    result.extend(right[j:])
    return result

// KEY INSIGHT

Keep dividing the array into halves until each part has one element. Then merge the sorted halves back together in order.

// FULL APPROACH

Merge sort works by recursively splitting the array into two halves, sorting each half, and merging the sorted halves. The merge step compares the smallest remaining values from both halves and builds the final sorted array.

def merge_sort_iterative(arr):
    n = len(arr)
    size = 1

    while size < n:
        for left in range(0, n, 2 * size):
            mid = min(left + size, n)
            right = min(left + 2 * size, n)
            arr[left:right] = merge(arr[left:mid], arr[mid:right])
        size *= 2

    return arr
